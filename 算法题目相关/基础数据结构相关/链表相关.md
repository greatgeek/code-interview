# 链表相关

## 链表反转

```go
type ListNode struct {
    Val  int
    Next *ListNode
}

// 反转尾节点为 nil 情况
func reverseList(head *ListNode) *ListNode {
    var prev *ListNode
    curr := head
    for curr != nil {
        curr.Next, prev, curr = prev, curr, curr.Next // 一次赋多值时，会将右边计算好同时赋给左边
    }
    return prev
}

// 反转尾节点不为 nil 情况
func myReverse(head, tail *ListNode) (*ListNode, *ListNode) {
	prev := tail.Next
	cur := head
	for prev != tail {
		cur.Next, cur, prev = prev, cur.Next, cur
	}
	return tail, head
}
```

## 快慢指针
```go
type ListNode struct {
    Val  int
    Next *ListNode
}

func findMiddle(head *ListNode) *ListNode {
    if head == nil {
        return nil
    }
    slow, fast := head, head
    for fast != nil && fast.Next != nil {
        slow, fast = slow.Next, fast.Next.Next
    }
    return slow
}
```
## [K 个一组翻转链表](https://leetcode.cn/problems/reverse-nodes-in-k-group/)

```go
/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func reverseKGroup(head *ListNode, k int) *ListNode {
	hair := &ListNode{Next: head}
	prev := hair

	for head != nil {
		tail := prev
		for i := 0; i < k; i++ {
			tail = tail.Next
			if tail == nil {
				return hair.Next
			}
		}
		next := tail.Next
		// 当前链表示结构为: prev->head->tail->next, 需要反转 head 和 tail 作为头和尾节点的链表
		head, tail = myReverse(head, tail)
		prev.Next, tail.Next = head, next
		prev = tail
		head = tail.Next
	}
	return hair.Next
}

func myReverse(head, tail *ListNode) (*ListNode, *ListNode) {
	prev := tail.Next
	cur := head
	for prev != tail {
		cur.Next, cur, prev = prev, cur.Next, cur
	}
	return tail, head
}
```

## [92. 反转链表 II](https://leetcode.cn/problems/reverse-linked-list-ii/)

```go
/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func reverseBetween(head *ListNode, left int, right int) *ListNode {
	dummy := &ListNode{}
	dummy.Next = head
	preStart, start, end := dummy, head, head
    for i := 0; i < left-1; i++ {
        // 步进到反转链表部分的前一个节点
        preStart = preStart.Next
    }
    // 反转链表部分的首节点，反转后将变成尾节点
    start = preStart.Next 
    // 反转链表部分的尾节点，反转后将变成首节点
    end = start 
    for i := 0; i < right-left; i++ {
        // 步进到反转链表部分的最后一个节点
        end = end.Next 
    }

	preStart.Next, end = reverse(start, end)
	return dummy.Next
}

func reverse(head, tail *ListNode) (*ListNode, *ListNode) {
	preNode := tail.Next
	curNode := head
	for preNode != tail {
		curNode, curNode.Next, preNode = curNode.Next, preNode, curNode
	}
	return tail, head
}
```
